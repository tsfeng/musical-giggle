#user#
CREATE TABLE `user` (
  `id` bigint(20) NOT NULL,
  `name` varchar(100) DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

#logger_info#
CREATE TABLE `logger_info` (
  `logger_id` bigint(20) NOT NULL AUTO_INCREMENT,
  `thread_name` varchar(100) DEFAULT NULL,
  `class_name` varchar(100) DEFAULT NULL,
  `method_name` varchar(50) DEFAULT NULL,
  `begin_time` varchar(50) DEFAULT NULL,
  `logger_level` varchar(10) DEFAULT NULL,
  `logger_name` varchar(100) DEFAULT NULL,
  `logger_message` text,
  `logger_throwable` text,
  PRIMARY KEY (`logger_id`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8;
